### Instructions

<InstructionsContainer>
  <InstructionsSection title="Prepare Ingredients">
    <InstructionStep stepNumber={1}>
      Wash and soak 2 <strong>cups</strong> <strong>basmati</strong> <strong>rice</strong> for 30 minutes
    </InstructionStep>

    <ServerStepImage stepNumber={1} alt="Basmati rice soaking in water" caption="Basmati rice soaking in water" />

    <InstructionStep stepNumber={2}>
      Cut 1 kg <strong>chicken</strong> <strong>into</strong> <strong>medium</strong> <strong>pieces</strong> and marinate with <strong>yogurt</strong> and spices
    </InstructionStep>

    <ServerStepImage stepNumber={2} alt="Marinated chicken pieces in a bowl" caption="Marinated chicken pieces in a bowl" />

    <InstructionStep stepNumber={3}>
      Slice 4 <strong>large</strong> <strong>onions</strong> <strong>thinly</strong> and deep fry until golden brown
    </InstructionStep>

    <ServerStepImage stepNumber={3} alt="Golden brown fried onions on paper towel" caption="Golden brown fried onions on paper towel" />

    <InstructionStep stepNumber={4}>
      Boil 4 <strong>eggs</strong> and 4 <strong>potatoes</strong> separately until cooked
    </InstructionStep>

    <ServerStepImage stepNumber={4} alt="Boiled eggs and potatoes cooling down" caption="Boiled eggs and potatoes cooling down" />

    <InstructionStep stepNumber={5}>
      Prepare biryani <strong>masala</strong> by grinding whole spices
    </InstructionStep>

    <ServerStepImage stepNumber={5} alt="Fresh ground biryani masala powder" caption="Fresh ground biryani masala powder" />
  </InstructionsSection>
  <InstructionsSection title="Cook the Dish">
    <InstructionStep stepNumber={6}>
      Heat oil in heavy-bottomed pot and add whole spices
    </InstructionStep>

    <ServerStepImage stepNumber={6} alt="Whole spices sizzling in hot oil" caption="Whole spices sizzling in hot oil" />

    <InstructionStep stepNumber={7}>
      Add marinated <strong>chicken</strong> and cook until 70% done
    </InstructionStep>

    <ServerStepImage stepNumber={7} alt="Chicken cooking in the pot with spices" caption="Chicken cooking in the pot with spices" />

    <InstructionStep stepNumber={8}>
      In separate pot, boil water with whole spices and <strong>salt</strong>
    </InstructionStep>

    <ServerStepImage stepNumber={8} alt="Large pot of boiling spiced water" caption="Large pot of boiling spiced water" />

    <InstructionStep stepNumber={9}>
      Add <strong>soaked</strong> <strong>rice</strong> to boiling water and cook until 70% done
    </InstructionStep>

    <ServerStepImage stepNumber={9} alt="Rice cooking in boiling spiced water" caption="Rice cooking in boiling spiced water" />

    <InstructionStep stepNumber={10}>
      Layer the partially cooked <strong>rice</strong> over <strong>chicken</strong>
    </InstructionStep>

    <ServerStepImage stepNumber={10} alt="Rice being layered over chicken in pot" caption="Rice being layered over chicken in pot" />
  </InstructionsSection>
  <InstructionsSection title="Final Assembly and Dum Cooking">
    <InstructionStep stepNumber={11}>
      Sprinkle <strong>fried</strong> <strong>onions</strong>, <strong>mint</strong> <strong>leaves</strong>, and <strong>saffron</strong> <strong>soaked</strong> in <strong>milk</strong>
    </InstructionStep>

    <ServerStepImage stepNumber={11} alt="Garnishing the biryani with fried onions and herbs" caption="Garnishing the biryani with fried onions and herbs" />

    <InstructionStep stepNumber={12}>
      Add boiled <strong>eggs</strong> and <strong>potatoes</strong> between <strong>rice</strong> layers
    </InstructionStep>

    <ServerStepImage stepNumber={12} alt="Eggs and potatoes arranged in biryani layers" caption="Eggs and potatoes arranged in biryani layers" />

    <InstructionStep stepNumber={13}>
      Cover pot with aluminum foil, then place tight-fitting lid
    </InstructionStep>

    <ServerStepImage stepNumber={13} alt="Pot covered with foil and lid for dum cooking" caption="Pot covered with foil and lid for dum cooking" />

    <InstructionStep stepNumber={14}>
      Cook on high heat for 3 minutes, then reduce to lowest heat for 45 minutes
    </InstructionStep>

    <ServerStepImage stepNumber={14} alt="Biryani pot on low heat for dum cooking" caption="Biryani pot on low heat for dum cooking" />
  </InstructionsSection>
</InstructionsContainer>

### Notes

<RecipeNotes>
  <>Use aged basmati rice for best results</>
  <>Don't overcook the rice - it should be 70% done before layering</>
  <>Dum cooking is crucial - keep heat very low to avoid burning</>
  <>Let the biryani rest for 10 minutes before opening the lid</>
  <>Serve hot with raita and boiled eggs on the side</>
  <>Can be stored in refrigerator for up to 3 days</>
</RecipeNotes>
